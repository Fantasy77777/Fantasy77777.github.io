# 使用vuex

import Vue from 'vue'
import Vuex from 'vuex'

Vue.use(Vuex)

export default new Vuex.Store({
  // strict: process.env.NODE_ENV !== 'production', // 严格模式，不建议在发布环境开启
  state: { // 定义数据
  },
  getter: { // 拦截数据的获取
  },
  mutations: { // 同步修改state的方法
  },
  actions: { // 异步修改state的方法
  },
  modules: { // 模块
  }
})

  1. 通过mapState辅助函数
import { mapState } from 'vuex'
computed: {
  ...mapState({
    data1: state => state.data1, // 这里属性名一样
    data222: state => state.data2, // 这里用data222来获取data2的值
    data3 (state) { // 注意了，这里的this指向的是当前的局部状态，this不是指向vuex
      return this.comValue + state.data1
    }
  })
}

// 如果属性名跟state中的一样，data1对应state中的data1
// 那么可以用字符串数组的形式来获取，如下
computed: {
  ...mapState([
    'data1',
    'data2'
  ])
}

# 组件修改state的值

  1. 不建议直接修改，虽然生效，但是不利于维护，开启严格模式的话，浏览器会发出警告
// 如果vuex中没开启严格模式，是可以这样修改，但是不建议
// 因为到时很多页面都用到data1，都修改了data1，很难追踪
this.$store.state.data1 = newValue
1
2
3
  1. 通过mutation的方法，这也是官方说的：更改 Vuex 的 store 中的状态的唯一方法是提交 mutation。

注意

mutation 必须是同步函数

// 在mutation中定义了一个修改state的方法
state: {
  name: '大薪薪'
},
mutations: {
  changeName (state, newName) { // 注意mutations中参数只能是2个，一个是state，一个是传过来的参数，多个参数情况，要利用对象/数组的方式来传参
    state.name = newName
  }
}

页面调用mutation中的方法

// 方法一：直接commit
this.$store.commit('changeName', 'newName参数')

// 方法二：在methods中通过mapMutations映射的方式
methods: {
  // 方法名一样的时候
  ...mapMutations(['changeName']), // 将 `this.changeName()` 映射为 `this.$store.commit('changeName')`
  change () {
    this.changeName('newName参数') // 即可改变
  }

  // 用新的方法名代替changeName
  ...mapMutations({
    changeNameFn: 'changeName' // 将 `this.changeNameFn()` 映射为 `this.$store.commit('changeName')`
  }),
  change () {
    this.changeNameFn('newName参数') // 即可改变
  }
}

  1. 当修改state的操作是异步操作时，通过action来修改state，当然还是要结合mutation

定义好action方法

mutations: {
  change (state, param) {
    state.data1 = param
  }
},
actions: {
  actionFn ({ commit }, param) {
    setTimeout(() => {
      commit('change', param) // 在这里提交commit
    }, 2000)
  }
}

组件中使用actions里面的方法

// 方法一：通过dispatch
this.$store.dispatch('actionFn', '参数')

// 方法二： mapActions辅助函数来进行映射
import { mapActions } from 'vuex'
export default {
  // ...
  methods: {
    // 不改变函数名
    ...mapActions([
      'actionFn', // 将 `this.actionFn()` 映射为 `this.$store.dispatch('actionFn')`
    ]),

    // 用新的函数名
    ...mapActions({
      actionFnNew: 'actionFn' // 将 `this.actionFnNew()` 映射为 `this.$store.dispatch('actionFn')`
    })
  }
}

# vuex的数据比较庞大时就进行分模块










 














 


















const moduleA = {
  namespaced: true, // 开启了带命名空间的模块
  state: {
    a: '我是moduleA的a',
    b: '我是moduleA的b'
  },
  getters: {},
  mutations: {
    test (state, param) {
      console.log('moduleA-test')
    }
  },
  actions: {}
}

const moduleB = {
  namespaced: true, // 开启了带命名空间的模块
  state: {
    a: '我是moduleB的a',
    b: '我是moduleB的b'
  },
  getters: {},
  mutations: {
    test (state, param) {
      console.log('moduleB-test')
    }
  },
  actions: {}
}

export default new Vuex.Store({
  modules: {
    moduleA: moduleA,
    moduleB: moduleB
  }
})

store.state.moduleA // 访问的是moduleA的东西
store.state.moduleA.a // 打印的是：我是moduleA的a

store.state.moduleB // 访问的是moduleB的东西
store.state.moduleB.a // 打印的是：我是moduleB的a
