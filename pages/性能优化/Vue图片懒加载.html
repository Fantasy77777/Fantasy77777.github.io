# Vue中图片懒加载

# 安装 vue-lazyload 插件

yarn install vue-lazyload --save-dev
1

# 记得在main.js引入和使用

import VueLazyload from 'vue-lazyload'
Vue.use(VueLazyload) // 默认使用

// 自定义使用

Vue.use(VueLazyload, {
preLoad: 1.3, // 预加载高度比例
error: 'dist/error.png', // 图片路径错误时加载的error图片
loading: 'dist/loading.gif', // 预加载图片，就是正式图片还没加载的时候，先用loading代替
attempt: 1 // 尝试加载图片的数量
})

# 使用方式

<!-- 默认的没有懒加载 -->
<img :src="imgUrl">

<!-- 只需要将:src改成v-lazy -->
<img v-lazy="imgUrl">

# 坑

本人实践发现使用的时候有个要注意的地方

<!-- imgList -->
  <div v-for="(item, index) in imgList" :key="index">
    <img :key="index" v-lazy="item.src" class="img-css">
  </div>

# 当图片资源是静态资源的时候,要使用 require

data () {
  return {
    imgList: [
      { src: require('../../../assets/test.jpg') },
    ]
  }
},
