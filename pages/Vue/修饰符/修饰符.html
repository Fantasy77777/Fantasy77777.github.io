# .stop 阻止事件冒泡

<div @click="alert('div')">
  <button @click.stop="alert('button')">button</button>
</div>
<!-- 只输出button -->

# .prevent 阻止事件默认行为

<a href="https://www.baidu.com" @click.prevent="myFn">test</a>
<!-- 默认的href不会触发 -->

# .self 点击元素本身触发

<div class="blue" @click.self="alert('div')">
  <button @click="alert('button')">button</button>
</div>
<!-- 只有点击div区域才会触发，只点button不会冒泡 -->

# .once只触发一次

# .sync 父子组件prop双向绑定

<!-- 父组件 -->
<comA :syncValue.sync="testValue"></comA>

<!-- 子组件 -->
this.$emit('update:syncValue', newValue)

<!-- 注意update冒号后面不能有空格，这是个小坑 -->
