# 生成vue3.0项目

安装vue-next

yarn add vue-next

# vue2.x和vue3.0的区别

vue2.x中main.js是这样的

import Vue from "vue"
import App from "./App.vue"
import router from "./router"
import store from "./store"

new Vue({
  router,
  store,
  render: h => h(App)
}).$mount("#app")

# 一个template模板大概样子

<template>
  <div class="container">
    <p>count的值是：{{state.count}}</p>
    <p>count * 2的值是：{{state.double}}</p>
    <button @click="add"></button>
  </div>
</template>
<script>
import { reactive, computed } from "vue";

export default {
  setup() {
    const state = reactive({
      count: 0,
      double: computed(() => state.count * 2)
    })

    function add() {
      state.count++
    }

    // 最后将页面用到的数据和methods返回
    return {
      state,
      add
    }
  }
}
</script>
